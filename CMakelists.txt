cmake_minimum_required(VERSION 3.16)

project(StatSearch VERSION 1.0 DESCRIPTION "Get sports stats quickly." LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# --- SDL build options ---
set(SDL_SHARED OFF CACHE BOOL "Build SDL3 as a shared library" FORCE)
set(SDL_STATIC ON CACHE BOOL "Build SDL3 as a static library" FORCE)
add_subdirectory(vendor/SDL EXCLUDE_FROM_ALL)

add_subdirectory(StatSearchAPI)

# --- IMGUI sources ---
set(IMGUI_SOURCES
    vendor/imgui/imgui.cpp
    vendor/imgui/imgui_demo.cpp
    vendor/imgui/imgui_draw.cpp
    vendor/imgui/imgui_tables.cpp
    vendor/imgui/imgui_widgets.cpp
    vendor/imgui/backends/imgui_impl_sdl3.cpp
    vendor/imgui/backends/imgui_impl_sdlrenderer3.cpp
)

# --- Project sources ---
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS
    "${PROJECT_NAME}/src/*.cpp"
)

list(APPEND SOURCES ${IMGUI_SOURCES})

# --- Executable ---
add_executable(${PROJECT_NAME} ${SOURCES})

target_link_options(${PROJECT_NAME} PRIVATE)
target_compile_definitions(${PROJECT_NAME} PRIVATE SDL_MAIN_HANDLED)

target_link_libraries(${PROJECT_NAME} PRIVATE SDL3::SDL3-static ws2_32 StatSearchAPI)

target_include_directories(${PROJECT_NAME} PUBLIC
    ${PROJECT_SOURCE_DIR}/src
    vendor/SDL/include
    vendor/spdlog/include
    vendor/imgui
    vendor/cpp-httplib
    vendor/json

    StatSearchAPI/include
)
